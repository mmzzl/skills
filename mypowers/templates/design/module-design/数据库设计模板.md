#### 3.3.1 表 `[table_name]`（[表中文名]）

```sql
-- 表：example
-- 用途：[描述该表的业务用途]
-- 所属仓库：repo-example
-- 预估数据量：[行数]
-- 读写比例：[读:写 如 10:1]

CREATE TABLE example (
  id          VARCHAR(36)   NOT NULL                    COMMENT '[字段含义]',
  name        VARCHAR(100)  NOT NULL                    COMMENT '[字段含义]',
  status      ENUM('value1','value2','value3')
                             NOT NULL DEFAULT 'value1'  COMMENT '状态枚举：value1-含义1, value2-含义2, value3-含义3',
  config      JSON          NOT NULL DEFAULT '{}'       COMMENT '配置信息，JSON结构',
  created_at  DATETIME(3)   NOT NULL DEFAULT CURRENT_TIMESTAMP(3),
  updated_at  DATETIME(3)            ON UPDATE CURRENT_TIMESTAMP(3),

  PRIMARY KEY (id),
  INDEX idx_name       (name)       COMMENT '支持按名称搜索（LIKE 前缀匹配）',
  INDEX idx_status     (status)     COMMENT '支持按状态过滤列表',
  INDEX idx_created_at (created_at) COMMENT '支持按创建时间排序'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='[表中文名]';
```
